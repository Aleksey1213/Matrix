int Task4 () {
    /*
        Задача №4.
        Сформировать матрицу размером 10 х 10 значения элементов которой произведения индексов. 
    */
    
    int A[10][10];
    
    for (int i = 0; i < 10; i++) {
        for(int j = 0; j < 10; j++) {
            A[i][j] = i * j;
        }
    }
    
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            cout << A[i][j] << "\t";
        }
        cout << "\n";
    }
    
    return 0;
}

int Task1 () {
    /*
        Задача №1.
        Вычислить сумму отрицательных элементов массива если их более 3-х или найти произведение положительных элементов.
        
    */
    
   int n, m;
   int k = 0, sum = 0, mul = 1;
   
   cout << "Введите количество строк: ";
   cin >> n;
   
   cout << "Введите количество столбцов: ";
   cin >> m;
   
   int** A = new int*[n];
   
   for (int i = 0; i < n; i++) {
       A[i] = new int[m];
   }
   
   for (int i = 0; i < n; i++) {
       for (int j = 0; j < m; j++) {
           A[i][j] = rand() % 1000 - 200;
       }
   }
   
   for (int i = 0; i < n; i++) {
       for (int j = 0; j < m; j++) {
           cout << A[i][j] << "\t";
       }
       cout << endl;
   }
   
   for (int i = 0; i < n; i++) {
       for (int j = 0; j < m; j++) {
           if (A[i][j] < 0) {
               k+=1;
           }
       }
   }
   
   if(k > 3) {
       for (int i = 0; i < n; i++) {
           for (int j = 0; j < m; j++) {
               if (A[i][j] < 0) {
                   sum+=A[i][j];
               }
           }
       }
       
       cout << "Сумма отрицательных элементов:\t" << sum << endl;
   }
   else {
       for (int i = 0; i < n; i++) {
           for (int j = 0; j < m; j++) {
               if (A[i][j] > 0) {
                   mul = mul * A[i][j];
               }
           }
       }
       
       cout << "Произведение положитльных элементов:\t" << mul << endl;
   }
   
   return 0;
}

int Task3 () {
    int A[10][10];
    int buf = 0, max, min;
    
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            A[i][j] = rand();
        }
    }
    
    for (int i = 0; i < 10; i++) {
        for (int j = 1; j < 10; j++) {
            if (A[i][j] > A[i][j-1] and A[i]) {
                buf = A[i][j];
            }
        }
    }
    
    max = buf;
    
    for (int i = 0; i < 10; i++) {
        for (int j = 0; j < 10; j++) {
            if (A[i][j] < buf) {
                buf = A[i][j];
            }
        }
    }
    
    min = buf;
    
    cout << "Разность максимального и минимального элемета массива:\t" << max-min << endl; 
    
}
